	(此外，請留意 $(xxx) 裏頭 xxx 的語法，完全就跟之前 CSS 教的 Selector 一模一樣。)

                $("#report").text("line1<br>Line2");


// 220:
                $("#report").html("line1<br>Line2");


// 230:
                $("#report").append("line1<br>Line2");


// 240:
                var s = $("#report").text();
				alert(s);
				// 與 210 相較，有何差別?


// 250
                var s = $("#aTextField").attr("value");
                $("#report").append(s);


// 260
                var s = $("#aTextField").attr("value", "Hello!");
